🤖 10. Complete FTP Automated Analyzer (Production Ready)

#!/bin/bash
LOGFILE=${1:-/var/log/vsftpd.log}
XFERLOG=${2:-/var/log/xferlog}

echo "╔════════════════════════════════════════════════╗"
echo "║      FTP SECURITY FORENSICS REPORT             ║"
echo "╚════════════════════════════════════════════════╝"
echo -e "\n[🔥 TOP 15 ATTACKER IPs (Failed)]"
grep "FAIL LOGIN" $LOGFILE | grep -oE '([0-9]{1,3}\.){3}[0-9]{1,3}' | sort | uniq -c | sort -nr | head -15

echo -e "\n[👥 TOP 15 TARGETED USERNAMES]"
grep "FAIL LOGIN" $LOGFILE | sed -E 's/.*\[([^\]]+)\] FAIL LOGIN.*/\1/' | sort | uniq -c | sort -nr | head -15

echo -e "\n[✅ TOP 10 SUCCESSFUL LOGINS by IP]"
grep "OK LOGIN" $LOGFILE | grep -oE '([0-9]{1,3}\.){3}[0-9]{1,3}' | sort | uniq -c | sort -nr | head -10

echo -e "\n[⚠️ COMPROMISED IPS (Fail + Success)]"
comm -12 <(grep "FAIL LOGIN" $LOGFILE | grep -oE '([0-9]{1,3}\.){3}[0-9]{1,3}' | sort -u) <(grep "OK LOGIN" $LOGFILE | grep -oE '([0-9]{1,3}\.){3}[0-9]{1,3}' | sort -u) | head -10

echo -e "\n[📦 TOTAL FAILED ATTEMPTS] $(grep -c "FAIL LOGIN" $LOGFILE)"
echo "[🔓 TOTAL SUCCESSFUL LOGINS] $(grep -c "OK LOGIN" $LOGFILE)"
echo -e "\n[📤 EXFILTRATION RISK - Top Downloads]"
grep "i" $XFERLOG 2>/dev/null | awk '{print $9}' | sort | uniq -c | sort -nr | head -10
